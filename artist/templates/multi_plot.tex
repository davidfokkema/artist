% \usepackage{tikz}
% \usetikzlibrary{arrows,pgfplots.groupplots}
% \usepackage{pgfplots}
% \pgfplotsset{compat=1.3}
% \usepackage[detect-family]{siunitx}
% \usepackage[eulergreek]{sansmath}
% \sisetup{text-sf=\sansmath}
% \usepackage{relsize}
%
\pgfkeysifdefined{/artist/width}
    {\pgfkeysgetvalue{/artist/width}{\defaultwidth}}
    {\def\defaultwidth{ {{ width }} }}
%
{%- if height %}
\pgfkeysifdefined{/artist/height}
    {\pgfkeysgetvalue{/artist/height}{\defaultheight}}
    {\def\defaultheight{ {{ height }} }}
{%- endif %}
%
\begin{sansmath}
\begin{tikzpicture}[font=\sffamily]
\node[inner sep=0pt] (plot) {
    \begin{tikzpicture}[
            inner sep=.3333em,
            font=\sffamily,
            every pin/.style={inner sep=2pt, font={\sffamily\smaller}},
            every label/.style={inner sep=2pt, font={\sffamily\smaller}},
            every pin edge/.style={<-, >=stealth', shorten <=2pt},
            pin distance=2.5ex,
        ]
        \begin{groupplot}[
                xmode={{ xmode }},
                ymode={{ ymode }},
                width=\defaultwidth,
                {%- if height %}
                height=\defaultheight,
                {%- endif %}
                %
                xmin={ {{ limits.xmin }} },
                xmax={ {{ limits.xmax }} },
                ymin={ {{ limits.ymin }} },
                ymax={ {{ limits.ymax }} },
                %
                group style={rows={{ rows }},columns={{ columns }},
                             horizontal sep=4pt, vertical sep=4pt},
                %
                tick align=outside,
                max space between ticks=40,
                every tick/.style={},
                axis on top,
                %
                xtick=\empty, ytick=\empty,
                scaled ticks=false,
            ]
            {% for subplot in subplots %}
                {% set plot = subplot.plot %}
                \nextgroupplot[
                {%- if subplot.empty %}
                    group/empty plot
                {%- else %}
                    % Default: empty ticks all round the border of the
                    % multiplot
                    {%- if ticks.x is not none and subplot.plot.ticks.x is not none %}
                        {%- if rows == 1 %}
                            xtick={ {{ ticks.x | join(', ') }} },
                            xtick pos=both,
                            xticklabel=\empty,
                        {%- elif subplot.row == 0 %}
                            xtick={ {{ ticks.x | join(', ') }} },
                            % 'right' means 'top'
                            xtick pos=right,
                            xticklabel=\empty,
                        {%- elif subplot.row == rows - 1 %}
                            xtick={ {{ ticks.x | join(', ') }} },
                            % 'left' means 'bottom'
                            xtick pos=left,
                            xticklabel=\empty,
                        {%- endif %}
                    {%- endif %}
                    {%- if ticks.y is not none and subplot.plot.ticks.y is not none %}
                        {%- if columns == 1 %}
                            ytick={ {{ ticks.y | join(', ') }} },
                            ytick pos=both,
                            yticklabel=\empty,
                        {%- elif subplot.column == 0 %}
                            ytick={ {{ ticks.y | join(', ') }} },
                            ytick pos=left,
                            yticklabel=\empty,
                        {%- elif subplot.column == columns - 1 %}
                            ytick={ {{ ticks.y | join(', ') }} },
                            ytick pos=right,
                            yticklabel=\empty,
                        {%- endif %}
                    {%- endif %}
                    {%- if subplot.plot.ticks.x %}
                        xtick={ {{ subplot.plot.ticks.x | join(', ') }} },
                    {%- endif %}
                    {%- if subplot.plot.ticks.y %}
                        ytick={ {{ subplot.plot.ticks.y | join(', ') }} },
                    {%- endif %}
                    {%- if subplot.show_xticklabel %}
                        xticklabel={},
                    {%- endif %}
                    {%- if subplot.show_yticklabel %}
                        yticklabel={},
                    {%- endif %}
                    {%- if subplot.xticklabel_pos %}
                        xticklabel pos={{ subplot.xticklabel_pos }},
                    {%- endif %}
                    {%- if subplot.yticklabel_pos %}
                        yticklabel pos={{ subplot.yticklabel_pos }},
                    {%- endif %}
                    axis equal={{ plot.axis_equal | lower }},
                    %
                    title={ {{ plot.title }} },
                    xlabel={ {{ plot.xlabel }} },
                    ylabel={ {{ plot.ylabel }} },
                    {%- if subplot.plot.limits.xmin is not none %}
                        xmin={ {{ subplot.plot.limits.xmin }} },
                    {%- endif %}
                    {%- if subplot.plot.limits.xmax is not none %}
                        xmax={ {{ subplot.plot.limits.xmax }} },
                    {%- endif %}
                    {%- if subplot.plot.limits.ymin is not none %}
                        ymin={ {{ subplot.plot.limits.ymin }} },
                    {%- endif %}
                    {%- if subplot.plot.limits.ymax is not none %}
                        ymax={ {{ subplot.plot.limits.ymax }} },
                    {%- endif %}
                {%- endif %}
                ]

                {% if not plot.histogram2d_list and not plot.shaded_regions_list and not plot.plot_series_list %}
                    % Force something to plot (invisibly).  Otherwise, the
                    % axis will be reset and have a default range *and*
                    % labels.  PGFPLOTS does not allow empty plots.
                    \addplot[] coordinates {(0, 0)};
                {% endif %}

                {% for histogram in plot.histogram2d_list %}
                    % Draw histogram background, if needed
                    {%- if histogram.type in ['bw', 'reverse_bw'] %}
                        {%- if histogram.type == 'bw' %}
                            {%- set histogrambg = 'black' %}
                        {%- elif histogram.type == 'reverse_bw' %}
                            {%- set histogrambg = 'white' %}
                        {%- endif %}
                        {%- set x0 = histogram.x_edges[0] %}
                        {%- set x1 = histogram.x_edges[-1] %}
                        {%- set y0 = histogram.y_edges[0] %}
                        {%- set y1 = histogram.y_edges[-1] %}
                        \path[fill,{{ histogrambg }}]
                            (axis cs:{{ x0 }}, {{ y0 }})
                            rectangle (axis cs:{{ x1 }}, {{ y1 }});
                    {%- endif %}

                    % Draw non-empty bins in histogram
                    {%- for i in range(histogram.x_edges|length - 1) %}
                        {%- for j in range(histogram.y_edges|length - 1) %}
                            {%- set x0 = histogram.x_edges[i] %}
                            {%- set x1 = histogram.x_edges[i + 1] %}
                            {%- set x = histogram.x_centers[i] %}
                            {%- set y0 = histogram.y_edges[j] %}
                            {%- set y1 = histogram.y_edges[j + 1] %}
                            {%- set y = histogram.y_centers[j] %}
                            {%- set count = histogram.counts[i][j] %}
                            {%- set weight = (100 * count / histogram.max) | int %}

                            {%- if weight > 0 %}
                                {%- if histogram.type == 'area' %}
                                \draw[{{ histogram.style }},
                                      scale around={ {{ count / histogram.max }}:
                                                    (axis cs: {{ x }}, {{ y }})}]
                                    (axis cs:{{ x0 }}, {{ y0 }})
                                    rectangle (axis cs:{{ x1 }}, {{ y1 }});
                                {%- elif histogram.type in ['bw', 'reverse_bw'] %}
                                \path[{{ histogram.style }},fill,
                                    {%- if 'reverse' in histogram.type %}
                                        {%- set weight = 100 - weight %}
                                    {%- endif %}
                                      black!{{ 100 - weight }}]
                                    (axis cs:{{ x0 }}, {{ y0 }})
                                    rectangle (axis cs:{{ x1 }}, {{ y1 }});
                                {%- endif %}
                            {%- endif %}
                        {%- endfor %}
                    {%- endfor %}
                {% endfor %}

                {% for region in plot.shaded_regions_list %}
                    \addplot[draw=none, fill={{ region.color }}] coordinates {
                    {%- for x, y in region.data %}
                        ({{ x }}, {{ y }})
                    {%- endfor %}
                        } -- cycle;
                {% endfor %}

                {% for series in plot.plot_series_list %}
                    \addplot[{{ series.options }}] coordinates {
                    {%- for x, y, xerr, yerr in series.data %}
                        ({{ x }}, {{ y }})
                    {%- endfor %}
                    };
                {% endfor %}

                {% for line in plot.horizontal_lines %}
                    \draw[{{ line.options }}]
                        ({rel axis cs:0, 0} |- {axis cs:0, {{ line.value }} }) --
                        ({rel axis cs:1, 1} |- {axis cs:0, {{ line.value }} });
                {% endfor %}

                {% for line in plot.vertical_lines %}
                    \draw[{{ line.options }}]
                        ({rel axis cs:0, 0} -| {axis cs:{{ line.value }}, 0 }) --
                        ({rel axis cs:1, 1} -| {axis cs:{{ line.value }}, 0 });
                {% endfor %}

                {% for pin in plot.pin_list %}
                    {%- if pin.use_arrow %}
                        {%- set pin_option = 'pin' %}
                    {%- else %}
                        {%- set pin_option = 'label' %}
                    {%- endif %}
                    \node[coordinate,
                          {{ pin_option }}={{ pin.location }}:{ {{ pin.text }} }]
                        at (axis cs:{{ pin.x }}, {{ pin.y }}) {};
                {% endfor %}

                {% if plot.label %}
                    \node[{{ plot.label.style }},
                    {%- if plot.label.node_location != 'center' %}
                          {{ plot.label.node_location }}=2pt
                    {%- endif %}
                        ]
                        at (rel axis cs:{{ plot.label.x }},
                            {{ plot.label.y }})
                        { {{ plot.label.text }} };
                {% endif %}

            {% endfor %}
        \end{groupplot}
        \end{tikzpicture}
    };
    \node[below] at (plot.south) { {{ xlabel }} };
    \node[above, rotate=90] at (plot.west) { {{ ylabel }} };
    \end{tikzpicture}
\end{sansmath}
